    def Nonce.mk_nonce(time = nil)
      salt = CryptUtil::random_string(6, @@NONCE_CHRS)
      if time.nil?
        t = Time.now.getutc
      else
        t = Time.at(time).getutc
      end
      time_str = t.strftime(TIME_FMT)
      return time_str + salt
    end
