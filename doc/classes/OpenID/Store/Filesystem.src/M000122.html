      def filename_escape(s)
        s = '' if s.nil?
        filename_chunks = []
        s.split('').each do |c|
          if @@FILENAME_ALLOWED.index(c)
            filename_chunks << c
          else
            filename_chunks << sprintf("_%02X", c[0])
          end
        end
        filename_chunks.join("")
      end
