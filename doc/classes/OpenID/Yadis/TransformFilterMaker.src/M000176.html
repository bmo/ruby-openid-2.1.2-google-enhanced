      def apply_filters(endpoint)
        # Apply filter procs to an endpoint until one of them returns
        # non-nil.
        @filter_procs.each { |filter_proc|
          e = filter_proc.call(endpoint)
          if !e.nil?
            # Once one of the filters has returned an endpoint, do not
            # apply any more.
            return e
          end
        }

        return nil
      end
