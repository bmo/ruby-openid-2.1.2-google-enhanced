      def parse_extension_args(args)
        @preferred_auth_policies = []
        policies_str = args['preferred_auth_policies']
        if policies_str
          policies_str.split(' ').each{|uri|
            add_policy_uri(uri)
          }
        end

        max_auth_age_str = args['max_auth_age']
        if max_auth_age_str
          @max_auth_age = max_auth_age_str.to_i
        else
          @max_auth_age = nil
        end
      end
